DROP VIEW GET_PART_IMPORTADOS;
ALTER TABLE "NF" ALTER COLUMN "COD_PART" TYPE character varying(20);
CREATE OR REPLACE VIEW "GET_PART_IMPORTADOS" AS
SELECT "COD_PART" "CODIGO",CAST(UPPER("PARTICIPANTE") AS CHARACTER VARYING(100))"PARTICIPANTE",
CAST(EXTRACT(MONTH FROM "DT_DOC") AS INTEGER) "MES" FROM "NF"
GROUP BY "COD_PART","PARTICIPANTE",EXTRACT(MONTH FROM "DT_DOC");
