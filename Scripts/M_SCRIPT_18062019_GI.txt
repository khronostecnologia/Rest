ALTER TABLE "NF" ADD "COD_EMP" character varying(20);
ALTER TABLE "NF" ADD "EMPRESA" character varying(100);
DROP VIEW "GET_PART_IMPORTADOS";
ALTER TABLE "NF" ADD "MES" INTEGER;
ALTER TABLE "NF" ADD "ANO" INTEGER;
CREATE OR REPLACE VIEW "GET_EMP_IMPORTADOS" AS
SELECT "COD_EMP" "CODIGO",CAST(UPPER("EMPRESA") AS CHARACTER VARYING(100))"EMPRESA",
"MES","ANO" FROM "NF"
GROUP BY "COD_EMP","EMPRESA","MES","ANO";
