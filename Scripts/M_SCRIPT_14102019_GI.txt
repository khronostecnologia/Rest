CREATE OR REPLACE VIEW "GET_CONTRIBUINTE" AS
 SELECT 
   "CONTRIBUINTE"."ID" AS "CODIGO",
    "CONTRIBUINTE"."NOME" AS "RAZAOSOCIAL",
    "CONTRIBUINTE"."CNPJ",
	"CONTRIBUINTE"."IE",
    "CONTRIBUINTE"."LOGRADOURO",
    "CONTRIBUINTE"."MUNICIPIO",
    "CONTRIBUINTE"."UF",
	"CONTRIBUINTE"."NUMERO",
	"CONTRIBUINTE"."COMPLEMENTO",
	"CONTRIBUINTE"."BAIRRO",
	"CONTRIBUINTE"."CEP",
	"CONTRIBUINTE"."TELEFONE"
   FROM "CONTRIBUINTE";
